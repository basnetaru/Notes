WHILE N < > 0
R = N MOD 10
S = S + R ^ 3

8.WAP to input a string and count the total numbers of vowels.


DECLARE FUNCTION COUNT (S$)
CLS
INPUT “ENTER ANY STRING”; S$
PRINT “TOTAL NO. OF VOWELS= “; COUNT(S$)
END

FUNCTION COUNT (S$)
VC = 0
FOR I = 1 TO LEN(S$)
B$ = MID$(S$, I, 1)
C$ = UCASE$(B$)
IF C$ = “A” OR C$ = “E” OR C$ = “I” OR C$ = “O” OR C$ = “U” THEN
VC = VC + 1
END IF
NEXT I
COUNT = VC
END FUNCTION
